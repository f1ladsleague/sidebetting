function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Bets"); var round = e.parameter.round; var constructorOrName = e.parameter.constructor; var raceType = e.parameter.raceType; var driver = e.parameter.driver; var amount = e.parameter.amount; var userType = e.parameter.userType || "Constructor"; var mobile = e.parameter.mobile || ""; var timestamp = new Date(); var data = sheet.getDataRange().getValues(); var updated = false; for (var i = 1; i < data.length; i++) { var rowRound = data[i][1]; var rowUser = data[i][2]; var rowRaceType = data[i][3]; var rowUserType = data[i][8]; var rowMobile = data[i][9]; var isMatch = false; if (userType === "Constructor") { if (rowRound == round && rowUser == constructorOrName && rowRaceType == raceType && (rowUserType == "Constructor" || !rowUserType)) { isMatch = true; } } else { if (rowRound == round && rowUser == constructorOrName && rowRaceType == raceType && rowUserType == "Public" && rowMobile == mobile) { isMatch = true; } } if (isMatch) { sheet.getRange(i + 1, 1).setValue(timestamp); sheet.getRange(i + 1, 5).setValue(driver); sheet.getRange(i + 1, 6).setValue(amount); updated = true; break; } } if (!updated) { // This writes to columns A through J sheet.appendRow([timestamp, round, constructorOrName, raceType, driver, amount, "", "", userType, mobile]); } return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); }
